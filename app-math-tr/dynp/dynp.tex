from functools import wraps

def memo(func):
    cache = {}                                  
    @wraps(func)                                
    def wrap(*args):                            
        if args not in cache:
            print 'miss', args
            cache[args] = func(*args)
        else: print 'hit', args
        return cache[args]                      
    return wrap 

def rec_dag_sp(W, s, t):                        
    @memo                                       
    def d(u):
        print ('in u')
        if u == t: return 0                     
        return min(W[u][v]+d(v) for v in W[u])  
    return d(s)                                 

dist = rec_dag_sp(DAG, 'a', 'f')
print 'uzaklik=', dist
\end{lstlisting}
