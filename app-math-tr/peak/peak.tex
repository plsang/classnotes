from scipy.optimize import fmin
import pandas as pd
from numpy.linalg import *
df = pd.read_csv('oil1.csv',sep='\s*')

def hubbard(w):
    yfit = (2*w[0]) / (1+cosh( np.abs(w[1] * (df['year']-w[2])) ))
    diff = df['oil'] - yfit
    e=norm(diff)
    return e

v = fmin(hubbard, [2, 2, 1970], maxiter=100000, maxfun=10000)
print v
\end{minted}
